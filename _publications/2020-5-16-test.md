double vUehf_smallr(double r, double rN, double chi) {
  // nb: a part can be removed and done analyitcally
  auto a = (r / rN) * chi;
  auto b = std::exp(-chi) * (1.0 + chi);
  auto c = std::sinh(a);
  return a - b * c;
}
